import numpy as np
import matplotlib.pyplot as plt
import pprint

angle = np.linspace(0,2*np.pi,180)

functions = {
    1:"omnidirectionnel",
    2:"bidirectionnel",
    3:"cardiode",
    4:"hypercardioide",
    5:"supercardioide"
}

def menu():
    print("Choisissez un diagramme de directivité dans la liste ci-dessous :")
    print(30*"-")
    pprint.pp(functions)
    print(30*"-")
    the_choice = int(input("Indiquez le numéro du diagramme souhaité : "))
    choice(the_choice)

def choice(c):
    if c==1: #omni
        func = np.full(180,1)
        the_min = -9
    elif c==2: #bi  
        func = np.cos(angle)
    elif c==3: #cardio
        func = 1+np.cos(angle)
    elif c==4: #hypercardio
        func = 1+2*np.cos(angle)
    elif c==5: #supercardio
        func = 1+3*np.cos(angle)
    
    func=20*np.log10(np.abs(func)/np.max(func))
    
    try: the_min
    except NameError: the_min = np.min(func)
    
    ax=plt.subplot(111,polar=True)
    ax.set_title("Directivité de type {}\n".format(functions[c]))
    
    ax.plot(angle,func)
    ax.grid(color="green")
    ax.set_theta_zero_location("N")
    
    plt.fill_between(angle,the_min,func, alpha=0.2, color="red")
    
    plt.tight_layout()
    plt.savefig("4_1.png")
    plt.show() 

menu()
